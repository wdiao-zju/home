se <- function(x) sd(x) / sqrt(length(x))
df_stat <- iris %>%
  group_by(Species) %>%
  summarise_at(
    vars(Sepal.Length),
    c(mean = mean, se = se)
  )
df_stat

ggplot(df_stat, aes(Species, mean, fill = Species)) +
  geom_errorbar(
    aes(ymin = mean * 0.9, ymax = mean + se),
    width = 0.2, size = 0.3
  ) +
  geom_col(color = "black", size = 0.3) +
  theme_classic()
